                    }),
                    go: () => {
                        const url = buildUrl();
                        return runPlugins(d => {
                            if (d.type === 'jsonp') return ajax.jsonp(url);
                            if (d.type === 'script') return ajax.script(url);
                            return ajax.xhr(url, d.type);
                        });
                    },
                    xhr: (url, type) => new Promise((res, rej) => {
                        const xhr = new XMLHttpRequest();
                        const m = d.method || 'get';
                        if (cfg.csrf && ['post', 'put', 'delete', 'patch'].includes(m)) {
                            d.headers['X-CSRF-Token'] = d.token || u.cookies()['go_csrf'];
                        }
                        if (['post', 'put', 'patch'].includes(m) && d.data && !d.body) {
                            d.headers['Content-Type'] = d.headers['Content-Type'] || 'application/x-www-form-urlencoded';
                            d.body = d.headers['Content-Type'].includes('json')
                                ? JSON.stringify(d.data)
                                : Object.entries(d.data).map(([k, v]) => `${encodeURIComponent(k)}=${encodeURIComponent(v)}`).join('&');
                        }
                        xhr.open(m, url, true);
                        Object.entries(d.headers).forEach(([k, v]) => xhr.setRequestHeader(k, v));
                        xhr.onload = () => {
                            if (xhr.status >= 200 && xhr.status < 300) {
                                const r = type === 'json' ? JSON.parse(xhr.responseText) : xhr.responseText;
                                if (type === 'html' && d.target) d.target.forEach(el => {
                                    if (d.swap === 'outerHTML') el.outerHTML = r;
                                    else if (d.swap === 'append') el.innerHTML += r;
                                    else el.innerHTML = r;
                                });
                                ajax.trigger('success', r);
                                ajax.trigger(xhr.status, r);
                                res(r);
                            } else {
                                const err = new Error(`HTTP ${xhr.status}`);
                                ajax.trigger('error', err);
                                ajax.trigger(xhr.status, err);
                                rej(err);
                            }
                        };
                        xhr.onerror = () => {
                            const err = new Error('Network error');
                            ajax.trigger('error', err);
                            rej(err);
                        };
                        xhr.send(d.body);
                    }),
                    jsonp: url => new Promise((res, rej) => {
                        const cb = `go_cb_${Date.now()}`;
                        window[cb] = r => {
                            ajax.trigger('success', r);
                            document.head.removeChild(s);
                            delete window[cb];
                            res(r);
                        };
                        const s = document.createElement('script');
                        s.src = `${url}${url.includes('?') ? '&' : '?'}${d.jsonp || 'callback'}=${cb}`;
                        s.onerror = () => {
                            ajax.trigger('error', new Error('JSONP error'));
                            document.head.removeChild(s);
                            delete window[cb];
                            rej(new Error('JSONP error'));
                        };
                        document.head.appendChild(s);
                    }),
                    script: url => new Promise((res, rej) => {
                        const s = document.createElement('script');
                        s.src = url;
                        s.onload = () => {
                            ajax.trigger('success', {});
                            res({});
                        };
                        s.onerror = () => {
                            ajax.trigger('error', new Error('Script error'));
                            document.head.removeChild(s);
                            rej(new Error('Script error'));
                        };
                        document.head.appendChild(s);
                    })
                };

                function buildUrl() {
                    let url = d.url;
                    if (!d.cache) url += `${url.includes('?') ? '&' : '?'}nocache=${Date.now()}`;
                    if (d.data && !['post', 'put', 'patch'].includes(d.method || 'get')) {
                        url += `${url.includes('?') ? '&' : '?'}${Object.entries(d.data).map(([k, v]) => `${encodeURIComponent(k)}=${encodeURIComponent(v)}`).join('&')}`;
                    }
                    return url;
                }

                function runPlugins(fn) {
                    return d.plugins.reduce((p, n) => p.then(c => window.goPlugins[n]?.before?.(c) || c), Promise.resolve(d))
                        .then(fn)
                        .then(r => d.plugins.reduce((p, n) => p.then(r => window.goPlugins[n]?.after?.(r) || r), Promise.resolve(r)))
                        .catch(err => { client.flash.show(`AJAX Error: ${err.message}`, 'danger'); throw err; });
                }

                function loadPlugin(n, ok, err) {
                    const s = document.createElement('script');
                    s.src = `${cfg.pluginDir}/${n}.js`;
                    s.onload = () => window.goPlugins[n] ? ok(window.goPlugins[n]) : err(new Error(`Plugin ${n} failed`));
                    s.onerror = () => err(new Error(`Failed to load ${n}`));
                    document.head.appendChild(s);
                }

                return ajax;
            },

            h: (t, a = {}, ...c) => {
                const el = document.createElement(t);
                Object.entries(a).forEach(([k, v]) => {
                    if (k.startsWith('on') && typeof v === 'function') el.addEventListener(k.slice(2).toLowerCase(), v);
                    else if (k === 'go-b') {
                        el.setAttribute('go-b', v);
                        client.dom.on(el, 'input', () => client.store.set(v, el.value));
                        if (client.store.get(v)) el.value = client.store.get(v);
                    } else el.setAttribute(k, v);
                });
                c.forEach(child => {
                    if (typeof child === 'string' || typeof child === 'number') el.appendChild(document.createTextNode(child));
                    else if (child instanceof Node) el.appendChild(child);
                    else if (Array.isArray(child)) child.forEach(c => el.appendChild(c instanceof Node ? c : document.createTextNode(c)));
                });
                return el;
            },

            c: class {
                constructor(p = {}) {
                    this.p = p;
                    this.s = {};
                    this.el = null;
                }
                set(s) { this.s = { ...this.s, ...s }; this.up(); }
                render() { return go.client.h('div', { class: 'go-card' }, 'Component'); }
                mount(e) { this.el = e; this.up(); this.init?.(); }
                up() { if (this.el) { this.el.innerHTML = ''; this.el.appendChild(this.render()); } }
            },

            icon: c => go.client.h('i', { class: `${c} go-icon` }),

            form: {
                serialize: f => Object.fromEntries(new FormData(f)),
                validate: (f, r) => {
                    const d = client.form.serialize(f);
                    const e = [];
                    for (const [k, v] of Object.entries(r)) {
                        if (v.required && !d[k]) e.push(`${k} required`);
                        if (v.min && d[k]?.length < v.min) e.push(`${k} must be >= ${v.min} chars`);
                        if (v.pattern && !new RegExp(v.pattern).test(d[k])) e.push(`${k} invalid`);
                    }
                    return e;
                }
            },

            store: {
                data: new Proxy({}, { set: debounce((t, k, v) => {
                    t[k] = v;
                    client.dom.all(`[go-b="${k}"], [go-t="${k}"], [go-show="${k}"]`).forEach(el => {
                        if (el.hasAttribute('go-b') && ['INPUT', 'TEXTAREA', 'SELECT'].includes(el.tagName)) el.value = v;
                        else if (el.hasAttribute('go-t')) el.textContent = v;
                        else if (el.hasAttribute('go-show')) el.style.display = v ? 'block' : 'none';
                    });
                    client.events.emit(`store:${k}`, v);
                    return true;
                }, 50) }),
                comp: new Map(),
                set: (k, v) => client.store.data[k] = v,
                get: k => client.store.data[k] || client.store.comp.get(k)?.() || null,
                compSet: (k, f) => client.store.comp.set(k, f),
                watch: (k, cb) => client.events.on(`store:${k}`, cb)
            },

            auth: {
                token: null,
                login: async (u, c) => {
                    try {
                        const r = await go.client.ajax().url(u).method('post').body(c).go();
                        client.auth.token = r.token;
                        sessionStorage.setItem('go_token', r.token);
                        client.events.emit('auth:login', r.token);
                        client.flash.show('Login OK', 'success');
                        return r;
                    } catch (e) {
                        client.flash.show(`Login failed: ${e.message}`, 'danger');
                        throw e;
                    }
                },
                logout: () => {
                    client.auth.token = null;
                    sessionStorage.removeItem('go_token');
                    client.events.emit('auth:logout');
                    client.flash.show('Logged out', 'success');
                },
                check: () => {
                    if (!client.auth.token) client.auth.token = sessionStorage.getItem('go_token');
                    return !!client.auth.token;
                }
            },

            flash: {
                queue: [],
                show: (m, t = 'info', d = 3000) => {
                    const a = go.client.h('div', { class: `go-alert go-alert-${t}`, role: 'alert' }, m);
                    document.body.appendChild(a);
                    client.flash.queue.push(a);
                    setTimeout(() => {
                        a.remove();
                        client.flash.queue = client.flash.queue.filter(x => x !== a);
                    }, d);
                }
            },

            events: {
                e: new Map(),
                on: (e, cb) => {
                    if (!client.events.e.has(e)) client.events.e.set(e, []);
                    client.events.e.get(e).push(cb);
                },
                emit: (e, ...a) => (client.events.e.get(e) || []).forEach(cb => cb(...a)),
                off: (e, cb) => client.events.e.set(e, (client.events.e.get(e) || []).filter(c => c !== cb))
            },

            i18n: {
                t: {
                    en: { welcome: 'Welcome', error: 'Error' },
                    es: { welcome: 'Bienvenido', error: 'Error' }
                },
                lang: l => cfg.lang = l,
                get: k => client.i18n.t[cfg.lang]?.[k] || k
            },

            comp: {
                c: new Map(),
                def: (n, t, d) => {
                    client.comp.c.set(n, { t, d });
                    client.dom.all(`[go-c="${n}"]`).forEach(el => {
                        const i = d();
                        el.innerHTML = u.tpl(t, i);
                        client.vdom.render(el, i);
                    });
                },
                render: () => client.comp.c.forEach((c, n) => client.dom.all(`[go-c="${n}"]`).forEach(el => {
                    const i = c.d();
                    el.innerHTML = u.tpl(c.t, i);
                    client.vdom.render(el, i);
                }))
            },

            plugin: (n, f) => {
                window.goPlugins = window.goPlugins || {};
                window.goPlugins[n] = f;
                f.init?.(go);
            },

            data: {
                each: (s, a, t) => {
                    const el = client.dom.get(s);
                    const d = client.store.get(a) || [];
                    el.innerHTML = d.map(i => u.tpl(t || el.innerHTML, i)).join('');
                },
                filter: (s, a, q) => {
                    const d = client.store.get(a) || [];
                    const f = d.filter(i => q.split(' in ').reduce((v, k) => v[k] || '', i).includes(q.split(' in ')[0]));
                    client.dom.get(s).innerHTML = f.map(i => u.tpl(client.dom.get(s).innerHTML, i)).join('');
                },
                sort: (s, a, k, o = 'asc') => {
                    const d = client.store.get(a) || [];
                    const srt = d.sort((x, y) => {
                        const v1 = x[k], v2 = y[k];
                        return o === 'asc' ? (v1 < v2 ? -1 : v1 > v2 ? 1 : 0) : (v1 > v2 ? -1 : v1 < v2 ? 1 : 0);
                    });
                    client.dom.get(s).innerHTML = srt.map(i => u.tpl(client.dom.get(s).innerHTML, i)).join('');
                }
            },

            log: m => cfg.debug && console.log(`[go.js] ${m}`),

            init: () => {
                client.css.inject();
                client.css.purge();
                client.router.handle();
                client.dom.all('[go-icon]').forEach(el => client.icons.render(el, el.getAttribute('go-icon')));
                client.dom.all('[go-on]').forEach(el => {
                    const [e, h] = el.getAttribute('go-on').split(':');
                    client.dom.on(el, e, () => window[h]?.());
                });
                client.dom.all('[go-b]').forEach(el => {
                    const k = el.getAttribute('go-b');
                    client.dom.on(el, 'input', () => client.store.set(k, el.value));
                    if (client.store.get(k)) el.value = client.store.get(k);
                });
                client.dom.all('[go-x]').forEach(el => {
                    const expr = el.getAttribute('go-x');
                    const v = u.evalExpr(expr, client.store.data);
                    if (v !== null) el.textContent = v;
                });
                client.dom.all('[go-ajax]').forEach(el => {
                    const [u, m = 'get', t = 'json', s = 'innerHTML'] = el.getAttribute('go-ajax').split(':');
                    const trg = el.getAttribute('go-target') || el;
                    const evt = el.getAttribute('go-trigger') || 'click';
                    client.dom.on(el, evt, () => client.ajax().url(u).method(m).type(t).target(trg).swap(s).go()
                        .catch(() => client.flash.show('Request failed', 'danger')));
                });
                client.dom.all('[go-each]').forEach(el => {
                    const [a, t] = el.getAttribute('go-each').split(':');
                    client.data.each(el, a, t);
                });
                client.dom.all('[go-filter]').forEach(el => {
                    const [a, q] = el.getAttribute('go-filter').split(':');
                    client.data.filter(el, a, q);
                });
                client.dom.all('[go-sort]').forEach(el => {
                    const [a, k, o] = el.getAttribute('go-sort').split(':');
                    client.data.sort(el, a, k, o);
                });
                client.dom.all('[go-init]').forEach(el => {
                    const f = el.getAttribute('go-init');
                    window[f]?.();
                });
                client.events.on('auth:login', () => client.router.go('/'));
                client.events.on('auth:logout', () => client.router.go('/login'));
                client.comp.render();
                client.log('Initialized');
            }
        };

        return { cfg, client, u };
    })();

    if (typeof window !== 'undefined') {
        window.go = go;
        go.client.init();
    }

    if (typeof module !== 'undefined' && module.exports) {
        module.exports = go;
    }
})();
